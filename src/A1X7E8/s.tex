\begin{theorem}
Let $f:[a,b]\to\RR$ be a continuous function on an interval 
$[a,b]\subset\RR$, and suppose that $f$ is differentiable 
on $(a,b)$, then there exists $\xi\in (a,b)$ such that
$$
\frac{f(b)-f(a)}{b-a}=f'(\xi).
$$

\end{theorem}
