\begin{theorem}
If $f:I\to\mathbb R$ on an interval $I\subseteq \RR$ is continuous 
at $x_0\in I$, then for any $\epsilon>0$ there exists $\delta>0$ 
(depending on $\epsilon$) such that 
$$
|f(x)-f(x_0)| < \epsilon
$$
for all $x\in I$ with $|x-x_0|<\delta$.
\end{theorem}
