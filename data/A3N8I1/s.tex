\begin{proposition}
Let $f: I\to\RR$ be an $n$-times differentiable function on an open 
interval $I\subseteq\RR$, then for any $a,b\in I$, there exists 
$\xi$ in between (i.e., $a<\xi<b$ or $b<\xi<a$) such that
$$
f(b) = \sum_{k=0}^{n-1} \frac{f^{(k)}(a)}{k!}(b-a)^k + 
\frac{f^{(n)}(\xi)}{n!}(b-a)^n
$$
\end{proposition}
